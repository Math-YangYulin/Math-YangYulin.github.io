& \leq M \sum_{T} \omega^{g}{i} \Delta x{i} + \left| \sum_{T} f(\xi_{i})g(\xi_{i}) \Delta x_{i} - \int_{a}^{b} f(x)g(x) \mathrm{d}x \right| \
& \leq M \sum_{T’’} {\omega^{g}{i}}’’ \Delta x{i}’’ + \left| \sum_{T} f(\xi_{i})g(\xi_{i}) \Delta x_{i} - \int_{a}^{b} f(x)g(x) \mathrm{d}x \right| \
& < M \dfrac{\varepsilon}{2M} + \dfrac{\varepsilon}{2} = \varepsilon
\end{split}
\end{equation}
